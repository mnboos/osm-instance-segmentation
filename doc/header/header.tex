% Small TODO note
\newcommand{\todonote}[1]{%
    \marginpar{\textcolor{\todonotecolor}{ToDo}}%
    \colorbox{red}{\sffamily\textcolor{white}{#1}}%
    \PackageWarning{ToDo:}{#1!}%
}

% Large TODO note
\newcommand{\todo}[1]{%
    \marginpar{\textcolor{\todonotecolor}{ToDo}}%
    \colorbox{red}{\parbox{\linewidth}{\sffamily\textcolor{white}{#1}}}%

}
